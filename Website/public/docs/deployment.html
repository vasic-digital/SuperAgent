Production Deployment

Deploy SuperAgent in production environments with Docker, Kubernetes, or bare metal installations. This guide covers system requirements, installation, configuration, and security hardening.

Deployment Modes

Single-Node

Development, testing, small-scale production. 8 CPU cores, 16GB RAM, 100GB storage.

Multi-Node

Medium-scale production. 3+ nodes, 4 CPU cores each, 8GB RAM each.

High-Availability

Large-scale production. 5+ nodes with load balancer and redundant storage.

Software Dependencies

# Install Go 1.21+
wget https://go.dev/dl/go1.21.5.linux-amd64.tar.gz
sudo tar -C /usr/local -xzf go1.21.5.linux-amd64.tar.gz
echo 'export PATH=$PATH:/usr/local/go/bin' >> ~/.bashrc
source ~/.bashrc

# Install Docker
curl -fsSL https://get.docker.com -o get-docker.sh
sudo sh get-docker.sh
sudo usermod -aG docker $USER

# Install Docker Compose
sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# Install PostgreSQL 14+
sudo apt update
sudo apt install postgresql postgresql-contrib

# Install Redis 7+
sudo apt install redis-server

Docker Deployment

The recommended way to deploy SuperAgent in production:

Docker Compose Configuration

version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    container_name: superagent-postgres
    environment:
      POSTGRES_DB: superagent
      POSTGRES_USER: superagent
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    networks:
      - superagent-network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U superagent"]
      interval: 10s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    container_name: superagent-redis
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data
    ports:
      - "6379:6379"
    networks:
      - superagent-network
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 5

  superagent:
    build: .
    container_name: superagent
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
    environment:
      - DB_HOST=postgres
      - DB_PASSWORD=${DB_PASSWORD}
      - REDIS_HOST=redis
      - CLAUDE_API_KEY=${CLAUDE_API_KEY}
      - DEEPSEEK_API_KEY=${DEEPSEEK_API_KEY}
    ports:
      - "8080:8080"
      - "8443:8443"
    networks:
      - superagent-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/health"]
      interval: 30s
      timeout: 10s
      retries: 3

volumes:
  postgres_data:
  redis_data:

networks:
  superagent-network:
    driver: bridge

Start Services

# Start core services
docker-compose up -d

# Start with AI services (Ollama)
docker-compose --profile ai up -d

# Start with monitoring (Prometheus, Grafana)
docker-compose --profile monitoring up -d

# Start everything
docker-compose --profile full up -d

# View logs
docker-compose logs -f superagent

Kubernetes Deployment

For large-scale production deployments with high availability:

Namespace and ConfigMap

apiVersion: v1
kind: Namespace
metadata:
  name: superagent
  labels:
    name: superagent
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: superagent-config
  namespace: superagent
data:
  config.yaml: |
    server:
      host: 0.0.0.0
      port: 8080
      mode: production
    database:
      host: postgres-service
      port: 5432
      name: superagent
    cache:
      type: redis
      host: redis-service
      port: 6379

Deployment and Service

apiVersion: apps/v1
kind: Deployment
metadata:
  name: superagent
  namespace: superagent
spec:
  replicas: 3
  selector:
    matchLabels:
      app: superagent
  template:
    metadata:
      labels:
        app: superagent
    spec:
      containers:
      - name: superagent
        image: superagent:latest
        ports:
        - containerPort: 8080
        envFrom:
        - secretRef:
            name: superagent-secrets
        volumeMounts:
        - name: config
          mountPath: /etc/superagent
        resources:
          requests:
            memory: "512Mi"
            cpu: "500m"
          limits:
            memory: "2Gi"
            cpu: "2000m"
        livenessProbe:
          httpGet:
            path: /health
            port: 8080
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /ready
            port: 8080
          initialDelaySeconds: 5
          periodSeconds: 5
      volumes:
      - name: config
        configMap:
          name: superagent-config
---
apiVersion: v1
kind: Service
metadata:
  name: superagent-service
  namespace: superagent
spec:
  selector:
    app: superagent
  ports:
  - port: 8080
    targetPort: 8080
  type: LoadBalancer

Systemd Service

For bare metal deployments using systemd:

[Unit]
Description=SuperAgent AI Orchestration Platform
Documentation=https://github.com/superagent/superagent
After=network.target postgresql.service redis-server.service
Wants=postgresql.service redis-server.service

[Service]
Type=simple
User=superagent
Group=superagent
WorkingDirectory=/opt/superagent
ExecStart=/opt/superagent/superagent --config /etc/superagent/config.yaml
ExecReload=/bin/kill -HUP $MAINPID
KillMode=mixed
TimeoutStopSec=30

# Environment
Environment=SUPERAGENT_CONFIG_PATH=/etc/superagent
EnvironmentFile=/etc/superagent/.env

# Security
NoNewPrivileges=true
PrivateTmp=true
ProtectSystem=strict
ProtectHome=true
ReadWritePaths=/var/log/superagent /var/lib/superagent

# Resource limits
LimitNOFILE=65536
LimitNPROC=4096
MemoryMax=4G
CPUQuota=400%

[Install]
WantedBy=multi-user.target

Enable and Start

# Copy service file
sudo cp superagent.service /etc/systemd/system/

# Reload systemd
sudo systemctl daemon-reload

# Enable and start
sudo systemctl enable superagent
sudo systemctl start superagent

# Check status
sudo systemctl status superagent

# View logs
sudo journalctl -u superagent -f

Security Hardening

TLS Configuration

# Generate self-signed certificates (development)
sudo openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
  -keyout /etc/superagent/certs/server.key \
  -out /etc/superagent/certs/server.crt \
  -subj "/C=US/ST=State/L=City/O=Organization/CN=superagent.local"

# Set permissions
sudo chmod 600 /etc/superagent/certs/server.key
sudo chmod 644 /etc/superagent/certs/server.crt

# For production, use Let's Encrypt
sudo certbot certonly --standalone -d your-domain.com

Firewall Configuration

# Configure UFW
sudo ufw allow 8080/tcp   # API
sudo ufw allow 8443/tcp   # HTTPS API
sudo ufw allow 5432/tcp   # PostgreSQL (internal only)
sudo ufw allow 6379/tcp   # Redis (internal only)
sudo ufw enable

# Configure fail2ban
sudo apt install fail2ban
sudo tee /etc/fail2ban/jail.local > /dev/null << 'EOF'
[superagent]
enabled = true
port = 8080,8443
filter = superagent
logpath = /var/log/superagent/error.log
maxretry = 5
bantime = 3600
findtime = 600
EOF

sudo systemctl restart fail2ban

Security Best Practices: Always use TLS in production, rotate API keys regularly, and enable audit logging.
